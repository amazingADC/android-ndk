# Build one shared lib. When we build, we set the bin to $user/tmp, and
# remove it after build complete, so only the distribution place could be
# used for application linking.

cmake_minimum_required(VERSION 3.4.1)

set(CMAKE_VERBOSE_MAKEFILE on)

# add 2 dependencies into build
add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/libwebp
    libwebp)

add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/ndk_helper
    ndk_helper)


add_library(native SHARED src/native.c)

target_link_libraries(native ndk-helper webp)

# copy out the lib binary
set(distribution_DIR ${CMAKE_CURRENT_SOURCE_DIR}/../../../../distribution)
set(targetFile ${distribution_DIR}/native/${ANDROID_ABI})
add_custom_command(TARGET native POST_BUILD
                   COMMAND "${CMAKE_COMMAND}" -E
                   copy "$<TARGET_FILE:native>" "${targetFile}/$<TARGET_FILE_NAME:native>"
                   COMMENT "Copying native to output directory")
